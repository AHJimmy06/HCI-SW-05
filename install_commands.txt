# 1. LIMPIEZA (Ejecutar estos comandos en la terminal dentro de usability-dashboard)
# Borra instalaciones previas y el archivo de bloqueo de versiones
Remove-Item -Path node_modules -Force -Recurse -ErrorAction SilentlyContinue
Remove-Item -Path package-lock.json -Force -ErrorAction SilentlyContinue

# 2. INSTALACIÓN DE DEPENDENCIAS DE PRODUCCIÓN
# Incluye Supabase, Enrutamiento, Iconos y utilidades de shadcn/ui
npm install @supabase/supabase-js react-router-dom lucide-react clsx tailwind-merge class-variance-authority @radix-ui/react-slot tslib

# 3. INSTALACIÓN DE DEPENDENCIAS DE DESARROLLO
# Incluye Vite, TypeScript, Tailwind CSS v4 y PostCSS
npm install -D vite @vitejs/plugin-react typescript @types/react @types/react-dom @types/node tailwindcss @tailwindcss/postcss postcss autoprefixer tailwindcss-animate

# 4. INICIAR EL PROYECTO (Forzando la reconstrucción del caché)
npm run dev -- --force
