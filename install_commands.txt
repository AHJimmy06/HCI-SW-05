# 1. LIMPIEZA (Ejecutar estos comandos en la terminal dentro de usability-dashboard)
# Borra instalaciones previas y el archivo de bloqueo de versiones
Remove-Item -Path node_modules -Force -Recurse -ErrorAction SilentlyContinue
Remove-Item -Path package-lock.json -Force -ErrorAction SilentlyContinue

# 2. INSTALACIÓN DE DEPENDENCIAS DE PRODUCCIÓN
# Incluye Supabase, Enrutamiento, Iconos, utilidades de UI y Generación de PDF
npm install @supabase/supabase-js react-router-dom lucide-react clsx tailwind-merge class-variance-authority @radix-ui/react-slot tslib jspdf jspdf-autotable

# 3. INSTALACIÓN DE DEPENDENCIAS DE DESARROLLO
# Incluye Vite, TypeScript, Tailwind CSS v4, PostCSS y Minificadores
npm install -D vite @vitejs/plugin-react typescript @types/react @types/react-dom @types/node tailwindcss @tailwindcss/postcss postcss autoprefixer tailwindcss-animate terser @types/jspdf

# 4. INICIAR EL PROYECTO (Forzando la reconstrucción del caché)
npm run dev -- --force

# 5. CONSTRUCCIÓN PARA PRODUCCIÓN (Verificado WCAG 2.2 / Lighthouse)
npm run build
